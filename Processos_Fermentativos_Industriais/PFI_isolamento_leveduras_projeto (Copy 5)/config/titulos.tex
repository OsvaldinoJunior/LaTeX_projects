% ================================================================
%  config/titulos.tex
%  Formato dos títulos conforme norma ABNT / disciplina.
%
%  Nível 1 (section):      MAIÚSCULO + NEGRITO
%  Nível 2 (subsection):   SÓ MAIÚSCULO
%  Nível 3 (subsubsection):minúsculo + negrito
%  Nível 4 (paragraph):    minúsculo + normal
%
%  Todos alinhados à margem esquerda.
%  Uma linha em branco após cada título.
% ================================================================

% Espaçamentos: {recuo}{espaço antes}{espaço depois}
\titlespacing*{\section}      {0pt}{1\baselineskip}{1\baselineskip}
\titlespacing*{\subsection}   {0pt}{1\baselineskip}{1\baselineskip}
\titlespacing*{\subsubsection}{0pt}{1\baselineskip}{1\baselineskip}
\titlespacing*{\paragraph}    {0pt}{1\baselineskip}{1\baselineskip}

% Seção primária – MAIÚSCULO e NEGRITO
\titleformat{\section}
    {\bfseries}
    {\thesection\space}
    {0pt}
    {\MakeUppercase}

% Seção secundária – SÓ MAIÚSCULO (sem negrito)
\titleformat{\subsection}
    {\normalfont}
    {\thesubsection\space}
    {0pt}
    {\MakeUppercase}

% Seção terciária – minúsculo e negrito
\titleformat{\subsubsection}
    {\bfseries}
    {\thesubsubsection\space}
    {0pt}
    {}

% Seção quaternária – minúsculo e normal
\titleformat{\paragraph}[hang]
    {\normalfont}
    {\theparagraph\space}
    {0pt}
    {}
